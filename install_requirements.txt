https://tinkerboarding.co.uk/wiki/index.php/1-wire

Setup the systemd service
sudo vi /etc/systemd/system/keezer.service

[Unit]
Description=Keezer Service
After=network.target
StartLimitIntervalSec=0

[Service]
Type=simple
Restart=always
RestartSec=1
User=linaro
ExecStartPre=
ExecStart=/usr/bin/python3 /home/linaro/keezer/keezer.py
ExecStartPost=
ExecStop=
ExecReload=

[Install]
WantedBy=multi-user.target

sudo systemctl daemon-reload
sudo systemctl enable keezer
sudo service keezer start
